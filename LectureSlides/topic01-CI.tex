% Slide 2: Introduction & Intuition
\begin{frame}
    \frametitle{Introduction \& Intuition}
    \begin{itemize}
        \item \textbf{The Problem:} We know the asymptotic distribution of an estimator $\hat{\theta}$, but we want to know the asymptotic distribution of a continuous function $g(\hat{\theta})$.
        \item \textbf{The Solution:} The Delta Method uses a first-order Taylor series expansion to approximate the variance of the transformed estimator.
        \item \textbf{Taylor Expansion:} Around the true parameter $\theta$, the function $g(\hat{\theta})$ can be approximated as:
        \begin{equation*}
            g(\hat{\theta}) \approx g(\theta) + g'(\theta)(\hat{\theta} - \theta)
        \end{equation*}
        \item Rearranging terms, we get:
        \begin{equation*}
            g(\hat{\theta}) - g(\theta) \approx g'(\theta)(\hat{\theta} - \theta)
        \end{equation*}
    \end{itemize}
\end{frame}

% Slide 3: The Univariate Delta Method
\begin{frame}
    \frametitle{The Univariate Delta Method}
    \begin{block}{Theorem}
        Let $X_n$ be a sequence of random variables such that:
        \begin{equation*}
            \sqrt{n}(X_n - \theta) \xrightarrow{d} \mathcal{N}(0, \sigma^2)
        \end{equation*}
        Suppose $g$ is a function such that its derivative $g'(\theta)$ exists and is non-zero. Then:
        \begin{equation*}
            \sqrt{n}(g(X_n) - g(\theta)) \xrightarrow{d} \mathcal{N}\left(0, \sigma^2 \left[g'(\theta)\right]^2\right)
        \end{equation*}
    \end{block}
    \vspace{0.5cm}
    \textbf{Implication:} The variance of the transformed estimator is approximately the original variance multiplied by the square of the derivative of the transformation evaluated at $\theta$.
\end{frame}

% Slide 4: Example - Survival Probability
\begin{frame}
    \frametitle{Example: Estimating Survival Probability}
    \begin{itemize}
        \item Let lifetimes $T_1, \dots, T_n \sim \text{Exp}(\lambda)$. The MLE for the rate is $\hat{\lambda} = 1/\bar{T}$.
        \item From asymptotic likelihood theory, we know:
        \begin{equation*}
            \sqrt{n}(\hat{\lambda} - \lambda) \xrightarrow{d} \mathcal{N}(0, \lambda^2)
        \end{equation*}
        \item We want the asymptotic distribution of the survival function at time $t_0$, defined as $S(t_0) = \exp(-\lambda t_0)$.
        \item Let $g(\lambda) = \exp(-\lambda t_0)$. The derivative is $g'(\lambda) = -t_0 \exp(-\lambda t_0)$.
        \item Applying the Delta Method:
        \begin{equation*}
            \sqrt{n}\left(\exp(-\hat{\lambda} t_0) - \exp(-\lambda t_0)\right) \xrightarrow{d} \mathcal{N}\left(0, \lambda^2 \left[-t_0 \exp(-\lambda t_0)\right]^2\right)
        \end{equation*}
        \item The asymptotic variance simplifies to $\lambda^2 t_0^2 \exp(-2\lambda t_0)$.
    \end{itemize}
\end{frame}

% Slide 5: Confidence Intervals
\begin{frame}
    \frametitle{Constructing Confidence Intervals}
    \begin{itemize}
        \item The asymptotic normality established by the Delta Method allows us to construct Wald-type confidence intervals for $g(\theta)$.
        \item We use a plug-in estimator for the standard error. Let $\widehat{\text{SE}}(g(\hat{\theta}))$ be:
        \begin{equation*}
            \widehat{\text{SE}}(g(\hat{\theta})) = \sqrt{\frac{\hat{\sigma}^2 \left[g'(\hat{\theta})\right]^2}{n}}
        \end{equation*}
        \item An approximate $(1 - \alpha)100\%$ confidence interval for $g(\theta)$ is given by:
        \begin{equation*}
            g(\hat{\theta}) \pm z_{\alpha/2} \cdot \widehat{\text{SE}}(g(\hat{\theta}))
        \end{equation*}
        \item \textit{Where $z_{\alpha/2}$ is the $(1 - \alpha/2)$ quantile of the standard normal distribution and $\hat{\sigma}^2$ is a consistent estimator of $\sigma^2$.}
    \end{itemize}
\end{frame}

% Slide 6: CI for Survival Probability (NEW)
\begin{frame}
    \frametitle{Example: CI for Survival Probability}
    \begin{itemize}
        \item Returning to our survival example, the estimated survival probability is $\hat{S}(t_0) = \exp(-\hat{\lambda} t_0)$.
        \item Using the asymptotic variance $\lambda^2 t_0^2 \exp(-2\lambda t_0)$, the standard error is estimated by plugging in $\hat{\lambda}$:
        \begin{equation*}
            \widehat{\text{SE}}(\hat{S}(t_0)) = \sqrt{\frac{\hat{\lambda}^2 t_0^2 \exp(-2\hat{\lambda} t_0)}{n}} = \frac{\hat{\lambda} t_0 \exp(-\hat{\lambda} t_0)}{\sqrt{n}}
        \end{equation*}
        \item A $(1 - \alpha)100\%$ asymptotic confidence interval for $S(t_0)$ is:
        \begin{equation*}
            \exp(-\hat{\lambda} t_0) \pm z_{\alpha/2} \left[ \frac{\hat{\lambda} t_0 \exp(-\hat{\lambda} t_0)}{\sqrt{n}} \right]
        \end{equation*}
        \item \textit{Note: For small samples, this linear CI might yield bounds outside $[0, 1]$. In practice, a complementary log-log transformation is often used to restrict bounds.}
    \end{itemize}
\end{frame}

% Slide 7: The Multivariate Delta Method
\begin{frame}
    \frametitle{The Multivariate Delta Method}
    \begin{block}{Theorem}
        Let $\bm{X}_n$ be a sequence of random vectors in $\mathbb{R}^k$ such that:
        \begin{equation*}
            \sqrt{n}(\bm{X}_n - \bm{\theta}) \xrightarrow{d} \mathcal{N}_k(\bm{0}, \Sigma)
        \end{equation*}
        Let $g: \mathbb{R}^k \rightarrow \mathbb{R}^m$ be a function whose Jacobian matrix $\nabla g(\bm{\theta})$ (of size $m \times k$) is evaluated at $\bm{\theta}$. Then:
        \begin{equation*}
            \sqrt{n}(g(\bm{X}_n) - g(\bm{\theta})) \xrightarrow{d} \mathcal{N}_m\left(\bm{0}, \nabla g(\bm{\theta}) \Sigma \nabla g(\bm{\theta})^T\right)
        \end{equation*}
    \end{block}
    \vspace{0.2cm}
    \textit{Note: The matrix $\nabla g(\bm{\theta}) \Sigma \nabla g(\bm{\theta})^T$ is the asymptotic covariance matrix of the transformed vector.}
\end{frame}
