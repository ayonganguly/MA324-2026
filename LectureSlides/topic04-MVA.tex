\section{Principal Component Analysis}

\begin{frame}
   \frametitle{Principal Component Analysis}
   \begin{itemize}
      \item Concerned with explaining the variance-covariance structure of a set
         of variables through a few linear combinations of these variables. 
         \vspace{0.45cm}
      \item Objectives
         \vspace{0.45cm}
         \begin{itemize}
            \item Data Reduction.
               \vspace{0.25cm}
            \item Interpretation.
         \end{itemize}
         \vspace{0.45cm}
      \item Serves as intermediate steps in a larger study (Multiple Linear
         Regression, Cluster Analysis).
         \vspace{0.45cm}
      \item Wide application areas: including image analysis, finance, health
         sector, cryptography, data-privacy etc. 
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Population Principal Components}
   \begin{itemize}
      \item Let $X_1,X_2,...,X_p$ be $p$-random variables with variance-covariance matrix $\Sigma$.
         \vspace{0.25cm}	
      \item First we want to find a linear combination of
         $\boldsymbol{X}=(X_1,X_2,...,X_p)$, say
         $\boldsymbol{a_{1}'}\boldsymbol{X}$, such that
         $Var(\boldsymbol{a_{1}'}\boldsymbol{X})$ is maximum.
         \vspace{0.25cm}		
      \item Note that
         $Var(\boldsymbol{a_{1}'}\boldsymbol{X})=\boldsymbol{a_{1}'}\Sigma\boldsymbol{a_1}$.
         Now by multiplying $\boldsymbol{a_1}$ by a constant, we can increase
         $\boldsymbol{a_{1}'}\Sigma\boldsymbol{a_1}$ arbitrarily.
         \vspace{0.25cm}
      \item Therefore, a more precise aim is:
         \begin{equation*}
            \operatorname*{max}_{\boldsymbol{a_1}}
            Var(\boldsymbol{a_{1}'}\boldsymbol{X})=\operatorname*{max}_{\boldsymbol{a_1}}
            \boldsymbol{a_{1}'}\Sigma\boldsymbol{a_1},
         \end{equation*}
         subject to $\boldsymbol{a_{1}'}\boldsymbol{a_{1}}=1$
   \end{itemize}	
\end{frame}

\begin{frame}
   \frametitle{}
   \begin{itemize}
      \item Thus, $1^{st}$ principal component = linear combination
         $\boldsymbol{a_{1}'}\boldsymbol{X}$ which maximizes
         $Var(\boldsymbol{a_{1}'}\boldsymbol{X})$ subject to
         $\boldsymbol{a_{1}'}\boldsymbol{a_{1}}=1$
         \vspace{0.25cm}
      \item Next, we want to find another linear cobination, say
         $\boldsymbol{a_{2}'}\boldsymbol{X}$, such that
         $Var(\boldsymbol{a_{2}'}\boldsymbol{X})$ is maximum subject to
         $\boldsymbol{a_{2}'}\boldsymbol{a_2}=1$ and
         $Cov(\boldsymbol{a_{1}'}\boldsymbol{X},\boldsymbol{a_{2}'}\boldsymbol{X})
         = 0$.
         \vspace{0.25cm}
      \item We proceed in the following manner	
         \begin{itemize}
            \item $i^{th}$ principle component = linear combination
               $\boldsymbol{a_{i}'}\boldsymbol{X}$ that maximizes
               $Var(\boldsymbol{a_{i}'}\boldsymbol{X})$ subject to
               $\boldsymbol{a_{i}'}\boldsymbol{a_i}=1$ and
               $Cov(\boldsymbol{a_{k}'}\boldsymbol{X},\boldsymbol{a_{i}'}\boldsymbol{X})
               = 0$ for $k<i$.
         \end{itemize}
         \vspace{0.25cm}
      \item Notice that
         $Cov(\boldsymbol{a_{k}'}\boldsymbol{X},\boldsymbol{a_{i}'}\boldsymbol{X})
         = \boldsymbol{a_{i}'}\Sigma\boldsymbol{a_k}$, and
         $Var(\boldsymbol{a_{i}'}\boldsymbol{X}) =
         \boldsymbol{a_{i}'}\Sigma\boldsymbol{a_i}$.
   \end{itemize}	
\end{frame}


\begin{frame}
   \frametitle{Theorem: Maximization of Quadratic Forms}
   Let $B_{p \times p}$ be a symmetric non-negative definite matrix with
   eigen-values $\lambda_1 \geq \lambda_2 \geq ... \geq \lambda_p \geq 0$ and
   associated normalized eigen-vectors $\boldsymbol{e_1},
   \boldsymbol{e_2},...,\boldsymbol{e_p}$. Then,
   \begin{eqnarray}
      &&	\operatorname*{max}_{\boldsymbol{x} \neq \boldsymbol{0}}
      \frac{\boldsymbol{x}^{'}B\boldsymbol{x}}{\boldsymbol{x}^{'}\boldsymbol{x}}
      = \lambda_1 \left(=
      \frac{\boldsymbol{e_1}^{'}B\boldsymbol{e_1}}{\boldsymbol{e_1}^{'}\boldsymbol{e_1}}
      \right)\ \text{ attained at } \boldsymbol{x}=\boldsymbol{e_1}, \nonumber \\
      && \operatorname*{min}_{\boldsymbol{x} \neq \boldsymbol{0}}
      \frac{\boldsymbol{x}^{'}B\boldsymbol{x}}{\boldsymbol{x}^{'}\boldsymbol{x}} =
      \lambda_p \text{ attained at } \boldsymbol{x}=\boldsymbol{e_p} \nonumber
   \end{eqnarray}	
   Moreover,
   \begin{eqnarray}
      &&	\operatorname*{max}_{\boldsymbol{x} \neq \boldsymbol{0}, \boldsymbol{x}
      \perp \boldsymbol{e_1},...,\boldsymbol{e_{k-1}}}
      \frac{\boldsymbol{x}^{'}B\boldsymbol{x}}{\boldsymbol{x}\boldsymbol{x}^{'}}
      = \lambda_k \text{ attained at } \boldsymbol{x}=\boldsymbol{e_k}, k=1,2,...,p
      \nonumber
   \end{eqnarray}
\end{frame}

\begin{frame}
   \frametitle{Theorem }
   Let $\Sigma_{p \times p}$ be the variance-covariance matrix of
   $\boldsymbol{X}_{p \times 1} = (X_1,...,X_p)$. Let $\Sigma$ have the
   eigen-value-eigen-vector pair
   $(\lambda_1,e_1),(\lambda_2,e_2),...,(\lambda_p,e_p)$ where $\lambda_1 \geq
   \lambda_2 \geq ... \geq \lambda_p \geq 0$. Then the $i^{th}$ principal
   component is given by,
   \begin{equation*}
      Y_i=\boldsymbol{e_i}^{'}\boldsymbol{X},\ \ \ \ i=1,2,...,p.
   \end{equation*}
   With these choices $Var(Y_i)=\lambda_i$ and $Cov(Y_i,Y_j)=0\ for\ i \neq j.$
   Moreover, if some $\lambda_i$ are equal, the choices of $e_i$ are not unique,
   and hence $Y_i$ are not unique.
\end{frame}

\begin{frame}
   \frametitle{Theorem }
   \begin{equation*}
      \sum_{i=1}^p Var(X_i) = \sigma_{11}+\sigma_{22}+...+\sigma_{pp}=tr(\Sigma)
      = \lambda_1+\lambda_2+...+\lambda_p=\sum_{i=1}^p Var(Y_i).
   \end{equation*}
   \vspace{0.5cm}
   \begin{itemize}
      \item Therefore, the proportion of total variance explained by $i^{th}$
         principal component is $\frac{\lambda_i}{\lambda_1+...+\lambda_p}$,
         $i=1,2,...,p$.
         \vspace{0.5cm}
      \item $e_{ik}$ measures the importance of $X_k$ to the $i^{th}$ principal
         component.
         % \vspace{0.5cm}
      % \item Particularly, $e_{ik}$ is proportional to the correlation
      %    coefficient between $Y_i$ and $X_k$. 
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Theorem }
   If $Y_1=\boldsymbol{e_1}^{'}\boldsymbol{X},
   Y_2=\boldsymbol{e_2}^{'}\boldsymbol{X}, \cdots,
   Y_p=\boldsymbol{e_p}^{'}\boldsymbol{X}$ are the principal components
   obtained from the covariance matrix $\Sigma$, then the correlation
   coefficient between the $Y_i$ and $X_k$ is 
   \begin{equation*}
      \rho_{Y_i,X_k} = \frac{e_{ik}\sqrt{\lambda_i}}{\sqrt{\sigma_{kk}}}.
   \end{equation*}
\end{frame}

\begin{frame}
   \frametitle{Principal Components Obtained from Standardized Variables}
   \begin{eqnarray}
      &&	Z_i = \frac{X_i-\mu_i}{\sqrt{\sigma_{ii}}}, i=1,2,...,p \nonumber \\
      &\Leftrightarrow& \boldsymbol{Z} = \begin{pmatrix}
         Z_1\\ . \\ . \\ . \\
         Z_p 
      \end{pmatrix} = V^{-1}(\boldsymbol{X}-\boldsymbol{\mu}),\ where\
      V=diag(\sqrt{\sigma_{11}},\sqrt{\sigma_{22}},...,\sqrt{\sigma_{pp}})
      \nonumber
   \end{eqnarray}
\end{frame}

\begin{frame}
   \begin{itemize}
      \item $Var(\boldsymbol{Z}) = \rho_{p \times p}=\begin{pmatrix}
            \rho_{11}(=1) & . & . & . & \rho_{1p} \\ . & & & & . \\ . & & & & .
            \\ . & & & & . \\ \rho_{p1} & . & . & . & \rho_{pp}(=1)
         \end{pmatrix}$
      \item $\sum_{i=1}^p Var(Y_i) = p$
         \vspace{0.5cm}
      \item Proportion of standardized population variance due to $k^{th}$
         principal component $=\frac{\lambda_k}{p},\ k=1,2,...,p,$ where
         $\lambda_k$'s are the eigen values of $\rho_{p \times p}$.
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Summarizing Sample Variation using Principal Components}
   \begin{itemize}
      \item Let $\boldsymbol{x_1},\boldsymbol{x_2},...,\boldsymbol{x_n}$ denote
         $n$ independent draws from a $p$-dimensional population with mean
         vector $\boldsymbol{\mu}$ and variance-covariance matrix $\Sigma$.
         \vspace{0.5cm}
      \item The sample mean is $\overline{\boldsymbol{x}}$, the sample
         variance-covariance matrix $S$.
         \vspace{0.5cm}
      \item $1^{st}$ sample principal component $=$ Linear combination
         $\boldsymbol{a_{1}'}\boldsymbol{x_j}$ that maximizes the sample
         variance of $\boldsymbol{a_{1}'}\boldsymbol{x_j}$ subject to
         $\boldsymbol{a_{1}'}\boldsymbol{a_1} = 1.$ 
   \end{itemize}
\end{frame}

\begin{frame}
   \begin{itemize}
      \item $i^{th}$ sample principal component $=$ Linear combination
         $\boldsymbol{a_{i}'}\boldsymbol{x_j}$ that maximizes the sample
         variance of $\boldsymbol{a_{i}'}\boldsymbol{x_j}$ subject to
         $\boldsymbol{a_{i}'}\boldsymbol{a_i} = 1$ and zero sample correlation
         between $\boldsymbol{a_{i}'}\boldsymbol{x_j}$ and
         $\boldsymbol{a_k}^{'}\boldsymbol{x_j}$ for all $k<i$.
         \vspace{0.75cm}
      \item Note that sample variance of $\boldsymbol{a_{1}'}\boldsymbol{x_j}$
         is $\boldsymbol{a_{1}'} S \boldsymbol{a_1}$. The sample covariance
         between $\boldsymbol{a_{i}'}\boldsymbol{x_j}$ and
         $\boldsymbol{a_k}^{'}\boldsymbol{x_j}$ is
         $\boldsymbol{a_{i}'}S\boldsymbol{a_k}$. 
   \end{itemize}  
\end{frame}

\begin{frame}
   \frametitle{Theorem}
   Let $S$ be $p \times p$ sample variance-covariance matrix with eigenvalue-eigenvector
   pair $(\hat{\lambda}_1,\hat{e}_1),...,(\hat{\lambda}_p,\hat{e}_p)$ with
   $\hat{\lambda}_1 \geq \hat{\lambda}_2 \geq ... \geq \hat{\lambda}_p \geq 0$.
   Then $i^{th}$ sample principal component is given by
   \begin{equation*}
      \hat{y}_i = \hat{e}_i^{'}\boldsymbol{x} = \hat{e}_{i1}x_1+...+\hat{e}_{ip}x_p,\ i=1,2,...,p
   \end{equation*}
   \vspace{0.25cm}
   Also, sample variance of $\hat{y}_i$ is $\hat{\lambda}_i,\ i=1,2,...,p$ and sample
   covariance $\hat{y}_i$ and $\hat{y}_k$ is $0$, for $i \neq k$.
\end{frame}

\begin{frame}
    \frametitle{Choosing the Number of Principal Components ($k$)}
    \textbf{Objective:} Reduce dimensions from $p$ to $k$ ($k \ll p$) while
    retaining most of the original variation. 
    \vspace{0.2cm}
    
    \begin{enumerate}
       \item No definite answer.
        \item \textbf{Proportion of Variance:} Retain first $k$ components such that
           the cumulative variance explained is sufficiently high (typically
           $70\%$ to $90\%$).
        \item \textbf{Scree Plot (Elbow Method):} Plot the eigenvalues
           $\lambda_i$ (ordered largest to smallest) against their index $i$.
           Retain components up to the ``elbow'' point.
    \end{enumerate}
    \vspace{0.2cm}

\section{Cluster Analysis}

\begin{frame}
   \frametitle{Cluster Analysis}
   \begin{itemize}
      \item Cluster analysis or clustering is the task of \textbf{grouping a set
         of objects} in such a way that objects in the \textbf{same group}
         (called a cluster) are \textbf{more similar} (in some sense) to each
         other than to those in other groups (clusters).  
         \vspace{0.5cm}
      \item Clustering can therefore be formulated as a \textbf{multi-objective optimization} problem.
         \vspace{0.5cm}
      \item Cluster analysis as such is not an automatic task, but an
         \textbf{iterative process} of knowledge discovery or interactive
         multi-objective optimization that involves trial and failure.
         \vspace{0.5cm}
      \item The basic objective in cluster analysis is to \textbf{discover
         natural groupings} of the items (or variables).
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Similarity Measures : Understanding Proximity}
   \begin{itemize}
      \item In cluster analysis, we must first develop a quantitative scale on
         which to \textbf{measure the association (similarity)} between objects.
         \vspace{0.50cm}	
      \item To understand the "\textbf{closeness}" or "\textbf{similarity}"
         among clusters, there can be two different methods:
         \vspace{0.50cm}		
         \begin{itemize}
            \item \textbf{Distance Measure} : Distances and Similarity
               Coefficients for Pairs of Items.
               \vspace{0.25cm}
            \item \textbf{Association Measure} : Similarities and Association
               Measures for Pairs of Variables. 
         \end{itemize}
   \end{itemize}	

\end{frame}


\begin{frame}
   \frametitle{Distance Measure}
   Several distance measures between two $p$-dimensional observations,
   $\mathbf{x}^{'}=[x_1,...,x_p]$ and $\mathbf{y}^{'} = [y_1,...,y_p]$:
   \vspace{0.5cm}
   \begin{itemize}
      \item \textbf{Euclidean distance} between two observations:
         \begin{equation*}
            d(\mathbf{x},\mathbf{y}) =
            \sqrt{(\mathbf{x}-\mathbf{y})^{'}(\mathbf{x}-\mathbf{y})}.
         \end{equation*}
      \item \textbf{Mahalanobis distance} (Statistical Distance) between two
         observations:
         \begin{equation*}
            d(\mathbf{x},\mathbf{y}) =
            \sqrt{(\mathbf{x}-\mathbf{y})^{'}\mathbf{S}^{-1}(\mathbf{x}-\mathbf{y})}.
         \end{equation*}
         where $\mathbf{S}$ is the matrix of sample variance-covariance matrix.
         \vspace{0.5cm}
      \item \textbf{Minkowski} metric between two observations:
         \begin{equation*}
            d(\mathbf{x},\mathbf{y}) =
            \left[\sum_{i=1}^p|x_i-y_i|^{m}\right]^{\frac{1}{m}}.
         \end{equation*}
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Similarity Measure}
   \begin{itemize}
      \item Sometimes items cannot be represented by meaningful $p$-dim
         measurements
      \item In such cases, pair of items may be compared based on presence or
         absence of certain characteristics.
   \end{itemize}
   \begin{center}
      \includegraphics[page = 692, trim = 8cm 1cm 5cm 24cm, scale = 0.7,
      clip=true]{Applied multivariate statistical analysis_Johnson_Wichern-5Ed_2002.pdf}
   \end{center}
   \vspace*{-2em}

   \begin{flushright}
      {\tiny  Reference: Applied Multivariate Statistical Analysis by Johnson and Wichern.}
   \end{flushright}
   \begin{itemize}
      \item Now, one can consider distance between $x^\prime = [1\, 0\, 0\, 1\,
         1]$ and $y=[1\, 1\, 0\, 1\, 0]$ 
      \item The Euclidean distance is 2
      \item Lower (higher) distance $\implies$ more (less) similarity
      \item Drawback: 1-1 match and 0-0 match has same importance
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Cluster Creation}
   Now, in cluster analysis, the main aim is to \textbf{create the clusters}
   using any one of the two major techniques, namely
   \vspace{0.5cm}
   \begin{itemize}
      \item \textbf{Hierarchical Clustering} : which mainly  proceeds by either
         a series of successive mergers or a series of successive divisions. The
         two types of hierarchical clustering are :
         \begin{itemize}
            \item Agglomerative hierarchical methods
            \item Divisive hierarchical methods (Self Study)
         \end{itemize}
         \vspace{0.75cm}
      \item \textbf{Non-Hierarchical Clustering} : technique is mainly designed
         to group items, rather than variables, into a collection of $K$
         clusters. The most common methodology is:
         \begin{itemize}
            \item K-Means Method
         \end{itemize}
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Agglomerative Hierarchical Methods}
   \begin{itemize}
      \item This methodology of clustering, start with the individual objects.
         \vspace{0.5cm}
      \item Initially as many clusters as objects.
         \vspace{0.5cm}
      \item The most similar objects are first grouped, and these initial groups
         are merged according to their similarities.
         \vspace{0.5cm}
      \item As clustering progresses, similarity decreases, and all subgroups
         are eventually fused into a single cluster.
         \vspace{0.5cm}
      \item One of the most common method of Agglomerative Hierarchical Method
         is \textbf{\textit{Linkage Method}}.
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Algorithm for Agglomerative Hierarchical Clustering}
   Usually while doing an agglomerative clustering methodology for grouping of
   $\textbf{\textit{N}}$ objects, the below steps (or algorithm) is followed:
   \vspace{0.25cm}
   \begin{itemize}
      \item Starting with $\textbf{\textit{N}}$ clusters, each one containing a
         single entity and an $\textit{N} \times \textit{N}$ symmetric matrix of
         distances (or similarities) $\textbf{D} = {d_{ik}}$.
         \vspace{0.25cm}
      \item  The distance matrix for the nearest (most similar) pair of clusters
         are observed. Let the \textbf{distance between "most similar" clusters}
         $\textit{U}$ and $\textit{V}$ be $d_{UV}$
         \vspace{0.25cm}
      \item After merge, clusters $\textit{U}$ and $\textit{V}$ as one newly
         formed cluster $(\textit{UV})$, the entries in the distance matrix are
         updated:
         \vspace{0.25cm}
         \begin{itemize}
            \item \textbf{Deleting} the rows and columns corresponding to
               clusters $\textit{U}$ and $\textit{V}$.
               \vspace{0.25cm}
            \item \textbf{Adding} a row and column for the distances between
               newly formed cluster $\textit{(UV)}$ and the remaining clusters.
         \end{itemize} 
         \vspace{0.25cm}
      \item The above steps are \textbf{repeated} for a total of $N-1$ times.
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Linkage Method}
   In the above mentioned algorithm, different forms of metric
   $\mathbf{D}(d_{ik})$ gives rise to different types of linkages and hence
   different types of clustering methodologies. The three used, linkages are: 
   \vspace{0.5cm}
   \begin{itemize}
      \item \textbf{Single Linkage} : $d_{(UV)W} = min\{ d_{UW},d_{VW}\}$
         \vspace{0.25cm}
      \item \textbf{Complete Linkage} : $d_{(UV)W} = max\{ d_{UW},d_{VW}\}$
         \vspace{0.25cm}
      \item \textbf{Averagre Linkage} : $d_{(UV)W} =
         \frac{\sum_{i}\sum_{k}d_{ik}}{N_{(UV)}N_W}$, where $N_{(UV)}$ and $N_W$
         are the number of items in clusters $\mathit{(UV)}$ and $\mathit{W}$,
         and $d_{ik}$ is the distance between $i^{th}$ object of cluster
         $\mathit{(UV)}$ and $k^{th}$ object of cluster $\mathit{W}$.
   \end{itemize}
\end{frame}

\begin{frame}
   \frametitle{Non-Hierarchical Clustering Methods}
   \begin{itemize}
      \item These techniques are commonly designed to \textbf{group items,
         rather than variables}, into a collection of $\mathit{K}$ clusters.
         \vspace{0.45cm}
      \item The \textbf{number of clusters}, $\mathit{K}$, may either be
         \textbf{specified in advance or determined as part} of the clustering
         procedure.
         \vspace{0.45cm}
      \item The Nonhierarchical clustering methods usually can start from any one of the two points :
         \vspace{0.15cm}
         \begin{itemize}
            \item an \textbf{initial partition} of items into groups.
               \vspace{0.25cm}
            \item an \textbf{initial set of seed points}, which will form the main nuclei of clusters.
         \end{itemize}
         \vspace{0.45cm}
      \item One of the \textbf{unbiased way} to start the clustering procedure
         is to, \textbf{randomly select seed points} from among the items or to
         randomly partition the items into initial groups.
   \end{itemize}    
\end{frame}

\begin{frame}
   \frametitle{K-Means Clustering}
   K-means is used to describe an algorithm that \textbf{assigns each item} to
   the \textbf{cluster having the nearest centroid (mean)}. The process mainly
   comprises of three steps :
   \vspace{0.3cm}

   \begin{itemize}
      \item First of all, \textbf{partition} the items into $\mathit{K}$
         \textbf{initial} clusters. [Or, specify $K$ initial centroids (seed
         points)]
         \vspace{0.45cm}
      \item Now, for each of list of items, \textbf{assigning an item} to the
         \textbf{cluster} whose \textbf{centroid} (mean) is \textbf{nearest}.
         (It has to be observed, distance is usually computed using Euclidean
         distance with either standardized or unstandardized observations.).
         \vspace{0.45cm}
      \item  Further, \textbf{recalculate the centroid} for the cluster
         receiving the new item and also for the cluster losing the item.
         \vspace{0.45cm}
      \item The above two steps are \textbf{repeated until no further reassignments} take place.
   \end{itemize}
\end{frame}

	\begin{itemize}

\begin{frame}
	\frametitle{Canonical Correlation Analysis}
	\begin{itemize}
		\item Let $\boldsymbol{X}^{(1)}_{p \times 1}$ and $\boldsymbol{X}^{(2)}_{q \times 1}$ be two random vector.
		\vspace{0.25cm}
		\item We assume that $p \leq q$.
		\vspace{0.25cm}
		\item $E(\boldsymbol{X}^{(1)}) = \boldsymbol{\mu}^{(1)}$, $E(\boldsymbol{X}^{(2)}) = \boldsymbol{\mu}^{(2)}$.
		\vspace{0.25cm}
		\item $Var(\boldsymbol{X}^{(1)})=\Sigma_{11}$, $Var(\boldsymbol{X}^{(2)})=\Sigma_{22}$
		\vspace{0.25cm}
		\item $Cov(\boldsymbol{X}^{(1)},\boldsymbol{X}^{(2)}) = (\Sigma_{12})_{p \times q}$, $Cov(\boldsymbol{X}^{(2)},\boldsymbol{X}^{(1)}) = (\Sigma_{21})_{q \times p}$ \\ In this case $\Sigma_{12}=\Sigma_{21}^{'}$
	\end{itemize}
\end{frame}

\begin{frame}
	\begin{itemize}
		\item $(\boldsymbol{X})_{(p+q) \times 1} = \begin{pmatrix}
		\boldsymbol{X}^{(1)} \\ \boldsymbol{X}^{(2)}
		\end{pmatrix}$, $(\boldsymbol{\mu})_{(p+q) \times 1} = \begin{pmatrix}
		\boldsymbol{\mu}^{(1)} \\ \boldsymbol{\mu}^{(2)}
		\end{pmatrix}=E(\boldsymbol{X})$, $\Sigma_{(p+q) \times (p+q)} = \begin{pmatrix}
		\Sigma_{11} & \Sigma_{12} \\
		\Sigma_{21} & \Sigma_{22}
		\end{pmatrix}=Var(\boldsymbol{X})$
		\vspace{0.25cm}
		\item First we want to find $\boldsymbol{a}$ and $\boldsymbol{b}$ such that $Corr(U_1=\boldsymbol{a_{1}'}\boldsymbol{X}^{(1)},V_1=\boldsymbol{b_1}^{'}\boldsymbol{X}^{(2)})$ is maximum.
		\vspace{0.25cm}
		\item Note that, \begin{equation*}
			Corr(U_1,V_1)=\frac{\boldsymbol{a_{1}'}\Sigma_{12}\boldsymbol{b_1}}{\sqrt{\boldsymbol{a_{1}'}\Sigma_{11}\boldsymbol{a_1}}\sqrt{\boldsymbol{b_1}^{'}\Sigma_{22}\boldsymbol{b_1}}}
		\end{equation*}
	\end{itemize}   
\end{frame}

\begin{frame}
	\begin{itemize}
		\item We need to solve the optimization problem,
		\begin{eqnarray}
			&&	\operatorname*{max}_{\boldsymbol{a_1},\boldsymbol{b_1}}  \frac{\boldsymbol{a_{1}'}\Sigma_{12}\boldsymbol{b_1}}{\sqrt{\boldsymbol{a_{1}'}\Sigma_{11}\boldsymbol{a_1}}\sqrt{\boldsymbol{b_1}^{'}\Sigma_{22}\boldsymbol{b_1}}},\ subject\ to\ Var(U_1)=Var(V_1)=1 \nonumber
		\end{eqnarray}
		\vspace{0.25cm}
		\item The \textbf{first pair} of canonical variables is the \textbf{pair of linear combinations} $U_1$ and $V_1$ having \textbf{unit variance}, which \textbf{maximizes} the \textbf{correlation} between $U_1$ and $V_1$.
		\vspace{0.25cm}
		\item The $k^{th}$ pair of canonical variables is the \textbf{pair of linear combinations} $U_k$ and $V_k$ having \textbf{unit variance}, which \textbf{maximizes} the \textbf{correlation} between $U_k$ and $V_k$ among all choices \textbf{uncorrelated with the previou}s $(k-1)$ canonical variables.
	\end{itemize}    
\end{frame}

\begin{frame}
	\frametitle{Theorem}
   Assume that $\Sigma$ has full rank. The \textbf{first pair of canonical
   variables} are given by,
   $U_1=\boldsymbol{e_1}^{'}\Sigma_{11}^{-\frac{1}{2}}\boldsymbol{X}^{(1)}$,
   $V_1=\boldsymbol{f_1}^{'}\Sigma_{22}^{-\frac{1}{2}}\boldsymbol{X}^{(2)}$ and
   $Corr(U_1,V_1)=\rho_1^*$. \vspace{0.25cm} \\ The $k^{th}\ (k=1,2,...,p)$ pair
   of canonical variables are given by,
   $U_k=\boldsymbol{e_k}^{'}\Sigma_{11}^{-\frac{1}{2}}\boldsymbol{X}^{(1)}$,
   $V_k=\boldsymbol{f_k}^{'}\Sigma_{22}^{-\frac{1}{2}}\boldsymbol{X}^{(2)}$ and
   $Corr(U_k,V_k)=\rho_k^*$. \vspace{0.25cm} \\ Here $\rho_1^* \geq \rho_2^*
   \geq ... \geq \rho_p^*$ are the \textbf{eigen values} of
   $\Sigma_{11}^{-\frac{1}{2}}\Sigma_{12}\Sigma_{22}^{-1}\Sigma_{21}\Sigma_{11}^{-\frac{1}{2}}$
   and $\boldsymbol{e_1},...,\boldsymbol{e_p}$ are corresponding \textbf{eigen
   vectors}. The quantities $\rho_1^*,...,\rho_p^*$ are \textbf{also} $p$
   largest \textbf{eigen values} of the matrix,
   $\Sigma_{22}^{-\frac{1}{2}}\Sigma_{21}\Sigma_{11}^{-1}\Sigma_{12}\Sigma_{22}^{-\frac{1}{2}}$
   with corresponding \textbf{eigen vectors}
   $\boldsymbol{f_1},...,\boldsymbol{f_p}$.
\end{frame}



\section{Factor Analysis}
	
	\begin{frame}
		\frametitle{Factor Analysis : Motivation}
		\begin{itemize}
			\item Try and understand, the \textbf{covariance relationships} \textbf{among many variables} in terms of \textbf{a few} underlying, but \textbf{unobservable, random quantities} called \textbf{factors}. 
			\vspace{0.5cm}
			\item Suppose all variables \textbf{within} a particular \textbf{group} are \textbf{highly correlated} among
			themselves, but have relatively \textbf{small correlations} with variables in a \textbf{different group}.
			\vspace{0.5cm}
			\item Then it is conceivable that \textbf{each group of variables} represents a \textbf{single} underlying
			\textbf{construct, or factor}, that is responsible for the observed correlations.
			\vspace{0.5cm}
			\item For example, correlations from the \textbf{group of test scores} in classics, Assamese (Asamiya), English, mathematics,
			and music  may suggest an underlying ``\textbf{intelligence}'' factor. A \textbf{second group} of variables representing \textbf{physical-fitness scores}, may corresponds to another factor. 
		\end{itemize}
		
		
	\end{frame}
	
	
	\begin{frame}
\frametitle{Orthogonal Factor Model with $m$ common Factors}
\begin{itemize}
	\item $X$ be the observable random vector with $p$ components with mean $\mu$ and variance-covariance matrix $\Sigma$. 	
\begin{eqnarray}
		\underset{(p \times 1)}{\mathbf{X}} && =\underset{(p \times 1)}{\mathbf{\boldsymbol{\mu}}}+\underset{(p \times m)}{\mathbf{L}}\underset{(m \times 1)}{\mathbf{F}}+\underset{(p \times 1)}{\mathbf{\boldsymbol{\varepsilon}}} \nonumber \\
	\implies  \underset{(p \times 1)}{\mathbf{X}} - \underset{(p \times 1)}{\mathbf{\boldsymbol{\mu}}} && = \underset{(p \times m)}{\mathbf{L}}\underset{(m \times 1)}{\mathbf{F}}+\underset{(p \times 1)}{\mathbf{\boldsymbol{\varepsilon}}} \nonumber \\
		 \mu_{i} && = mean\ of\ variable\ i, i = 1, \cdots, p, \nonumber \\ 
		 \varepsilon_{i} && =\ i^{th}\ specific\ factor (errors), i = 1, \cdots, p, \nonumber \\ 
		 F_{j} && =\ j^{th}\ common\ factor, j = 1, \cdots, m, \nonumber \\
		 l_{ij} && =\ loading\ of\ the\ i^{th}\ variable\ on\ j^{th}\ factor.\nonumber \\
		 F && = [F_1, \cdots, F_m]' ;  L = [l_{ij}] \nonumber 
	\end{eqnarray}
%	\vspace{0.2cm}

	\begin{frame}
\frametitle{Covariance Structure for the Orthogonal Factor Model}
\begin{itemize}
	\item The unobservable random vectors $\mathbf{F}$ and $\mathbf{\boldsymbol{\varepsilon}}$ satisfy the following conditions:
	\vspace{0.1cm}		
	\begin{itemize}
		\item $\mathbf{F}$ and $\mathbf{\boldsymbol{\varepsilon}}$ are independent.
		\item $E(\mathbf{F})=\mathbf{0}$, $Cov(\mathbf{F})=\mathbf{I}$.
		\item $E(\mathbf{\boldsymbol{\varepsilon}}) = \mathbf{0}$, $Cov(\mathbf{\boldsymbol{\varepsilon}}) = \Psi$, where $\Psi$ is a \textbf{diagonal} matrix. 
		\item $Cov(\mathbf{\boldsymbol{\varepsilon}}, \mathbf{F} )= 0 $
	\end{itemize}
\vspace{0.25cm}
	\item $Cov(\mathbf{X}) = \mathbf{L}\mathbf{L}^{'}+\mathbf{\Psi}$ \\ or it can also be expressed as, \\ 
	$Var(X_i)=l_{i1}^2+l_{i2}^2+...+l_{im}^2+\psi_{i}$ \\ $Cov(X_i,X_k)=l_{i1}l_{k1}+...+l_{im}l_{km}.$
	\vspace{0.25cm}
	\item $Cov(\mathbf{X},\mathbf{F})=\mathbf{L}$ \\ or \\ $Cov(X_i,F_j) = l_{ij}.$
	
\end{itemize}
\end{frame}
	
	\begin{frame}
		\frametitle{Some Features}
		\begin{itemize}
			\item The model $\mathbf{X}-\boldsymbol{\mu} = \mathbf{L}\mathbf{F}+\mathbf{\boldsymbol{\varepsilon}}$ is linear in common factors.
			\vspace{0.45cm}
			\item The portion of the variance of the $i^{th}$ variable ($X_i$) contributed by the $m$ common factors is termed as the $i^{th}$ \textit{\textbf{communality}}.
			$$Var(X_i)= \sigma_{ii} = \underbrace{l_{i1}^2+l_{i2}^2+...+l_{im}^2}_{communality} + \underbrace{\psi_{i}}_{specific \ variance}$$
%			\vspace{0.45cm}
			\item Also, the portion of $Var(X_i)=\sigma_{ii}$ due to the specific factor ($\epsilon_i$) is called the \textit{\textbf{uniqueness}}, or \textit{\textbf{specific variance}}.
			\vspace{0.45cm}
			\item Also, it can be seen easily that $\sigma_{ii} = h_i^2+\psi_i$, $i=1,2,...,p$, where $h_i^2 = l_{i1}^2+l_{i2}^2+...+l_{im}^2$.
		\end{itemize}
	\end{frame}
	
\begin{frame}

\begin{itemize}
	\item If $m >1$, then there is always some \textbf{inherent ambiguity} associated with the factor model. Let $T$ be any $m \times m$ orthogonal matrix, so
	$$\mathbf{X}-\boldsymbol{\mu} = \mathbf{L}\mathbf{F}+\mathbf{\boldsymbol{\varepsilon}} = \mathbf{L}\mathbf{T}\mathbf{T'}\mathbf{F}+\mathbf{\boldsymbol{\varepsilon}} = \mathbf{L^*}\mathbf{F^{*'}}+\mathbf{\boldsymbol{\varepsilon}}, $$
	where $\mathbf{L^*} = \mathbf{L}\mathbf{T}$ and $\mathbf{F^{*'}} = \mathbf{T'}\mathbf{F} $
	\vspace{0.25cm}
	\item Factor loadings $\mathbf{L}$ are determined only up to an orthogonal matrix $\mathbf{T}$. Thus the loadings, \begin{equation*}
		\mathbf{L}^{*} = \mathbf{L}\mathbf{T}\ and\ \mathbf{L} 
	\end{equation*} \\  both give the same representation. The \textbf{communalities}, given by the diagonal elements of $\mathbf{L}\mathbf{L}^{'} = (\mathbf{L}^{*})(\mathbf{L}^{*})^{'}$ are also \textbf{unaffected} by the choice of $\mathbf{T}$.
		\vspace{0.25cm}
	\item The \textbf{analysis} of the factor model \textbf{proceeds by imposing conditions} that \textbf{allow} one to \textbf{uniquely estimate} $\mathbf{L}$ and $\Psi$. ($\mathbf{L}'\Psi^{-1}\mathbf{L} = \ diagonal \ matrix$).
\end{itemize}
\end{frame}
	
	\begin{frame}
		\frametitle{Methods of Estimation}
		\begin{itemize}
			\item The sample covariance matrix $\mathbf{S}$ is an estimator of the unknown population covariance matrix $\mathbf{\Sigma}$. 
			\vspace{0.5cm}
			\begin{itemize}
				\item If the \textbf{off-diagonal elements} of matrix $\mathbf{S}$ are \textbf{small} or those of the sample
				correlation matrix $\mathbf{R}$ essentially zero, the variables are not related, and a \textbf{factor analysis will not be useful} in that scenario. In these circumstances, the specific factors play the dominant role, whereas the \textbf{major aim} of factor analysis is to \textbf{determine a few important common factors}.
				\vspace{0.5cm}
				\item If $\mathbf{\Sigma}$ appears to \textbf{deviate significantly from any diagonal matrix}, then a factor model
				can be fitted, With the initial problem being \textbf{estimating} the \textbf{factor loadings} $l_{ij}$
				and \textbf{specific variances} $\psi_{i}$. 
			\end{itemize}
		\end{itemize}    
	\end{frame}
	
	\begin{frame}
		\begin{itemize}
			\item The two most popular methods for parameter estimation are:
			\vspace{0.75cm}
			\begin{itemize}
				\item Principal Component (or Principal Factor).
				\vspace{0.5cm}
				\item Maximum Likelihood (Self Study). 
			\end{itemize}
		\end{itemize}
	\end{frame}
	
   \begin{frame}
      \frametitle{Principal Component Solution of the Factor Model}
      \begin{itemize}
         \item The principal component factor analysis of the sample covariance
            matrix $\mathbf{S}$ is specified in terms of its
            eigenvalues-eigenvector pairs
            $(\hat{\lambda}_1,\hat{\mathbf{e}}_1),(\hat{\lambda}_2, 
            \hat{\mathbf{e}}_2),...,(\hat{\lambda}_p,\hat{\mathbf{e}}_p)$,
            where $\hat{\lambda}_1 \geq \hat{\lambda}_2 \geq \hat{\lambda}_3
            \geq ... \geq \hat{\lambda}_p$. Let $m<p$ be the number of common
            factors. Then the matrix of estimated factor loadings
            $\{\Tilde{l}_{ij}\}$ is obtained as,
            %			\begin{equation*}
            %				\Tilde{\mathbf{L}} = \left[\sqrt{\hat{\lambda}_1}\hat{\mathbf{e}}_1 \vdots \sqrt{\hat{\lambda}_2}\hat{\mathbf{e}}_2\vdots ... \vdots \sqrt{\hat{\lambda}_m}\hat{\mathbf{e}}_m \right]
            %			\end{equation*}
            \begin{equation*}
               % \delimitershortfall=0pt
               % \setlength{\dashlinegap}{2pt}
               \Tilde{\mathbf{L}} =
               \left[\begin{array}{c|c|c|c}
                     \sqrt{\hat{\lambda}_1}\hat{\mathbf{e}}_1 &
                     \sqrt{\hat{\lambda}_2}\hat{\mathbf{e}}_2 & \cdots &
                     \sqrt{\hat{\lambda}_m}\hat{\mathbf{e}}_m
                  \end{array}
               \right]
            \end{equation*}			
      \end{itemize}
   \end{frame}
	
	\begin{frame}
		\begin{itemize}
			\item The estimated specific variances are provided by the diagonal elements of the matrix $\mathbf{S}-\Tilde{\mathbf{L}}\Tilde{\mathbf{L}^{'}}$, so,
			\begin{equation*}
				\Tilde{\mathbf{\Psi}}=\left[
				\begin{array}{cccc}
					\Tilde{\psi}_1 & 0 & \cdots & 0 \\
					0 & \Tilde{\psi}_2 & \cdots & 0 \\
					\vdots & \vdots & \ddots & \vdots \\
					0 & 0 & \cdots & \Tilde{\psi}_p
				\end{array}
				\right],\ with\ \Tilde{\psi}_i=s_{ii}-\sum_{j=1}^m \Tilde{l}_{ij}^2
			\end{equation*}
			\vspace{0.25cm}
			\item \textbf{Communalities} are estimated as,
			\begin{equation*}
				\Tilde{h}_i^2 = \Tilde{l}_{i1}^2+\Tilde{l}_{i2}^2+...+\Tilde{l}_{im}^2.
			\end{equation*}
			\vspace{0.25cm}
			\item The principal component \textbf{factor analysis of the sample correlation matrix} is obtained by starting with $\mathbf{R}$ in place of $\mathbf{S}$.
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Maximum Likelihood Method}
		\begin{itemize}
			\item If the common factors $\mathbf{F}$ and the specific factors $\boldsymbol{\varepsilon}$ can be assumed to be normally
			distributed, then maximum likelihood estimates of the factor loadings and specific
			variances may be obtained.
			\vspace{0.45cm}
			\item When $\mathbf{F}_j$ and $\boldsymbol{\varepsilon}_j$ are jointly normal, the observations $\mathbf{X}_j-\boldsymbol{\mu} = \mathbf{L}\mathbf{F}_j+\boldsymbol{\varepsilon}_j$ are then normal.
			\vspace{0.45cm}
			\item The corresponding likelihood is,
			\begin{align*}
				\mathit{L}(\boldsymbol{\mu},\boldsymbol{\Sigma}) & = \left(2\boldsymbol{\pi}\right)^{-\frac{(n-1)p}{2}}|\boldsymbol{\Sigma}|^{-\frac{(n-1)}{2}}\mathbf{e}^{-\left(\frac{1}{2}\right)tr\left[\boldsymbol{\Sigma}^{-1}\left(\sum_{j=1}^{n} (x_j-\overline{x})(x_j-\overline{x})^{'}\right)\right]} \\ & \times \left(2\boldsymbol{\pi}\right)^{-\frac{p}{2}}|\boldsymbol{\Sigma}|^{-\frac{1}{2}}\mathbf{e}^{-\left(\frac{n}{2}\right)(\overline{\mathbf{x}}-\boldsymbol{\mu})^{'}\boldsymbol{\Sigma}^{-1}(\overline{\mathbf{x}}-\boldsymbol{\mu})}
			\end{align*}
		\end{itemize}    
	\end{frame}

	\begin{frame}
		\begin{itemize}
			\item The above likelihood depends on  $\mathbf{L}$ and $\boldsymbol{\Psi}$ through $\boldsymbol{\Sigma} = \mathbf{L}\mathbf{L}^{'}+\boldsymbol{\Psi}$.
			\vspace{0.45cm}
			\item $\mathbf{L}$ is made well defined by imposing the computationally convenient uniqueness condition by,
			\begin{equation*}
				\mathbf{L}^{'}\boldsymbol{\Psi}^{-1}\mathbf{L} = \boldsymbol{\Delta},\ a\ diagonal\ matrix.
			\end{equation*}
			\vspace{0.45cm}
			\item The maximum likelihood estimates $\mathbf{\hat{L}}$ and $\boldsymbol{\hat{\Psi}}$ must be obtained by numerical maximization of the above likelihood function.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Result}
		
		Let $\mathbf{X}_1,\mathbf{X}_2,...,\mathbf{X}_n$ be a random sample from $\mathit{N}_p\left(\boldsymbol{\mu},\boldsymbol{\Sigma}\right)$, where $\boldsymbol{\Sigma} = \mathbf{L}\mathbf{L}^{'}+\boldsymbol{\Psi}$ is the covariance matrix for the Orthogonal Factor Model with $m$ common factors. The
		maximum likelihood estimators $\mathbf{\hat{L}}$, $\boldsymbol{\hat{\Psi}}$, and $\boldsymbol{\hat{\mu}}=\mathbf{\overline{x}}$ maximizes the above likelihood function subject to the condition, $\mathbf{L}^{'}\boldsymbol{\Psi}^{-1}\mathbf{L}$ being diagonal. \\ \vspace{0.25cm} The maximum likelihood estimates of the communalities are, \begin{equation*}
			\hat{h}_i^2=\hat{l}_{i1}^2+\hat{l}_{i2}^2+...+\hat{l}_{im}^2,\ for\ i=1,2,...,p
		\end{equation*},\\ \vspace{0.25cm} so, it can be said, 
		\begin{equation*}
			Proportion\ of\ total\ sample\ variance\ due\ to\ j^{th}\ factor = \frac{\hat{l}_{1j}^2+\hat{l}_{2j}^2+...+\hat{l}_{pj}^2}{s_{11}+s_{22}+...+s_{pp}}
		\end{equation*}
		
	\end{frame}
